library(dplyr)
results <- list.files('c:/users/darpa2/downloads',
                      pattern = 'november.*general',
                      ignore.case = T,
                      full.names = T)
results <- lapply(results,
                  data.table::fread,
                  col.names = tolower)
results <- data.table::rbindlist(results,
                                 fill = T)


henrico <- results |> 
  filter(localityname == 'HENRICO COUNTY') |> 
  mutate(officetitle = gsub(' -.*', '', officetitle),
         officetitle = ifelse(officetitle == 'United States Senate',
                              'Member United States Senate',
                              officetitle))
unique(henrico$officetitle)

# 2018 Ballot Standards
# https://www.elections.virginia.gov/media/formswarehouse/election-management/ballots/2019-07-25-SBE-Ballot-Standards-and-Verification-Procedures.pdf

ballot_order_henrico <- data.frame(
  office = c('President and Vice President',
             'Member United States Senate',
             'Member House of Representatives',
             'Governor',
             'Lieutenant Governor',
             'Attorney General',
             'Member Senate of Virginia',
             'Member House of Delegates',
             'Clerk of Court',
             "Commonwealth's Attorney",
             'Sheriff',
             'Member Board of Supervisors',
             'Member School Board',
             'Soil and Water Conservation Director Henricopolis District'),
  order = 1:14,
  type = c(rep('state', 6),
           rep('regional', 2),
           rep('locality', 3),
           rep('district', 3)),
  short_office = c('Pres. & VP',
                   'US Senate',
                   'US House',
                   'Gov.',
                   'L.G.',
                   'A.G.',
                   'VA Senate',
                   'VA House',
                   'Clerk',
                   'Comm. Att.',
                   'Sheriff',
                   'BoS',
                   'Sch. B.',
                   'Soil/Water')
)
henrico <- henrico |> 
  left_join(ballot_order_henrico, by = c('officetitle' = 'office')) |> 
  mutate(officetitle = factor(officetitle,
                              ordered = T,
                              levels = ballot_order_henrico$office))
totals <- henrico |> 
  mutate(total_votes = ifelse(grepl('Soil', officetitle),
                              total_votes/3,
                              total_votes)) |> 
  group_by(electiondate, order, officetitle, precinctname) |> 
  summarize(total_votes = sum(total_votes, na.rm = T)) |> 
  group_by(electiondate, precinctname) |> 
  mutate(vote_pct = total_votes / max(total_votes, na.rm = T))

total_summ <- totals |> 
  group_by(electiondate, officetitle) |> 
  summarise(med = median(vote_pct))

library(ggplot2)

ggplot() +
  geom_point(data = totals,
             aes(x = officetitle, y = vote_pct*100), color = 'lightgray') +
  geom_line(data = total_summ,
            aes(x = officetitle, y = med*100, group = electiondate)) +
  scale_x_discrete(labels = ballot_order_henrico$short_office) +
  facet_wrap(~electiondate) +
  labs(x = NULL, y = 'Voters (%)') +
  theme_bw() +
  theme(axis.text.x = element_text(angle = 90, hjust = 1, vjust = -1),
        panel.grid.minor.y = element_blank())
